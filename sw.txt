Code (simple DBSCAN):
from sklearn.cluster import DBSCAN
import numpy as np

def detect_clusters(data):
    coords = np.array([[d["lat"], d["lon"]] for d in data])

    if len(coords) < 3:
        return []

    clustering = DBSCAN(eps=0.01, min_samples=2).fit(coords)

    labels = clustering.labels_

    clusters = []
    for i, label in enumerate(labels):
        if label != -1:
            clusters.append({
                "lat": data[i]["lat"],
                "lon": data[i]["lon"],
                "cluster": int(label)
            })

    return clusters
